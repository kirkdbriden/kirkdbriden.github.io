Here is the code for the file:


    
    

    def get_file():
        filepath = input("What is the name of the file? ")
        return filepath
    
    def read_data(filepath):
        with open(filepath) as file:
            lines = file.readlines()[1:]
            return lines
    
    def create_array(lines):
        array = []
        for line in lines:   
            items_list = []
            items = line.split(",")
            for item in items:
                items_list.append(item)
            array.append(items_list)
        return array
    
    def get_crypto_symbol(array):
        crypto_symbol = array[0][2]
        return crypto_symbol
    
    def get_crypto_name(array):
        crypto_name = array[0][1]
        return crypto_name
    
    def get_max_closing_price(array):
        closing_prices = []
        for line in array:
            closing_prices.append(line[7])
        max_closing_price = max(closing_prices)
        max_closing_price = float(max_closing_price)
        return max_closing_price
    
    def get_max_closing_price_date(array, max_closing_price):
         for line in array:
             if float(line[7]) == max_closing_price:
                 max_cp_date = line[3]
         max_cp_date = max_cp_date[:10]
         return max_cp_date
    
    def get_min_closing_price(array):
        closing_prices = []
        for line in array:
            closing_prices.append(line[7])
        min_closing_price = min(closing_prices)
        return float(min_closing_price)
    
    def get_min_closing_price_date(array, min_closing_price):
         for line in array:
             if float(line[7]) == min_closing_price:
                 min_cp_date = line[3]
         min_cp_date = min_cp_date[:10]
         return min_cp_date
    
    
    def get_max_market_cap(array):
        market_caps = []
        for line in array:
            market_caps.append(line[9])
        max_market_cap = max(market_caps)
        return float(max_market_cap)
    
    def get_max_market_cap_date(array, max_market_cap):
         for line in array:
             if float(line[9]) == max_market_cap:
                 max_mc_date = line[3]
         max_mc_date = max_mc_date[:10]
         return max_mc_date
    
    def get_average_opening_price(array):
        opening_prices = []
        for line in array:
            opening_prices.append(float(line[6]))
        avg_opening_price = sum(opening_prices)/len(opening_prices)
        return avg_opening_price
        
    def get_average_closing_price(array):
        closing_prices = []
        for line in array:
            closing_prices.append(float(line[7]))
        avg_closing_price = sum(closing_prices)/len(closing_prices)
        return avg_closing_price
    
    
    def main():
        filepath = get_file()
        lines = read_data(filepath)
        array = create_array(lines)
        max_closing_price = get_max_closing_price(array)
        max_cp_date = get_max_closing_price_date(array, max_closing_price)
        min_closing_price = get_min_closing_price(array)
        min_cp_date = get_min_closing_price_date(array, min_closing_price)
        max_market_cap = get_max_market_cap(array)
        max_mc_date = get_max_market_cap_date(array, max_market_cap)
        avg_opening_price = get_average_opening_price(array)
        avg_closing_price = get_average_closing_price(array)
        print("File Name:", filepath)
        print("Crypto currency symbol:", get_crypto_symbol(array))
        print("Crypto currency name:", get_crypto_name(array))
        print("Highest closing price: $", '{:.2f}'.format(round(max_closing_price, 2)), " on ", max_cp_date, sep="")
        print("Lowest closing price: $", '{:.2f}'.format(round(min_closing_price, 2)), " on ", min_cp_date, sep="")
        print("Highest market cap: $", '{:.2f}'.format(round(max_market_cap, 2)), " on ", max_mc_date, sep="")
        print("Average opening price: $", '{:.2f}'.format(round(avg_opening_price, 2)), sep="")
        print("Average closing price: $", '{:.2f}'.format(round(avg_closing_price, 2)), sep="")
        print("Number of entries: ", len(lines))
    
    
    main()
